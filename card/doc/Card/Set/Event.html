def event event, stage_or_opts={}, opts={}, &final
  if stage_or_opts.is_a? Symbol
    opts[:in] = stage_or_opts
  else
    opts = stage_or_opts
  end
  process_stage_opts opts

  Card.define_callbacks event
  define_event event, opts, &final
  set_event_callbacks event, opts
end
